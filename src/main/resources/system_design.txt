药品信息 (Drug Information)
    drug_id: 唯一标识药品的ID
    name: 药品名称
    manufacturer_id: 关联的制造商ID
    batch_number: 批号
    expiry_date: 有效期
    type_id: 关联的药品类型ID
    current_version: 当前版本号
    create_date: 创建日期
    last_update_date: 最后更新日期

药品信息历史 (Drug Information History)
    history_id: 唯一标识历史记录的ID
    drug_id: 关联的药品ID
    version: 版本号
    name: 药品名称
    manufacturer_id: 关联的制造商ID
    batch_number: 批号
    expiry_date: 有效期
    type_id: 关联的药品类型ID
    modified_date: 修改日期
    modified_by: 修改人（可关联用户表）
    change_description: 修改描述
    create_date: 创建日期
    last_update_date: 最后更新日期

药品类型 (Drug Type)
    type_id: 唯一标识药品类型的ID
    type_name: 药品类型名称
    description: 类型描述
    create_date: 创建日期
    last_update_date: 最后更新日期

制造商信息 (Manufacturer Information)
    manufacturer_id: 唯一标识制造商的ID
    name: 制造商名称
    address: 制造商地址
    contact_info: 联系信息
    country: 国家
    established_date: 成立日期
    create_date: 创建日期
    last_update_date: 最后更新日期

运输信息 (Shipment Information)
    shipment_id: 唯一标识运输的ID
    drug_id: 关联的药品ID
    from_location: 出发地
    to_location: 目的地
    departure_date: 出发日期
    arrival_date: 到达日期
    carrier_id: 关联的承运商ID
    status: 运输状态
    tracking_number: 追踪号码
    create_date: 创建日期
    last_update_date: 最后更新日期

存储信息 (Storage Information)
    storage_id: 唯一标识存储的ID
    drug_id: 关联的药品ID
    location_id: 关联的存储地点ID
    temperature: 存储温度
    humidity: 存储湿度
    storage_date: 存储日期
    capacity: 存储容量
    status: 存储状态
    create_date: 创建日期
    last_update_date: 最后更新日期

销售信息 (Sales Information)
    sale_id: 唯一标识销售的ID
    drug_id: 关联的药品ID
    retailer_id: 关联的零售商ID
    sale_date: 销售日期
    quantity: 销售数量
    price: 销售价格
    discount: 折扣信息
    create_date: 创建日期
    last_update_date: 最后更新日期
    
区块链交易信息 (Blockchain Transaction Information)
    transaction_id: 区块链交易ID
    timestamp: 交易时间戳
    previous_hash: 前一个区块的哈希
    current_hash: 当前区块的哈希
    data: 交易数据
    transaction_type: 交易类型
    create_date: 创建日期
    last_update_date: 最后更新日期

-- 药品信息表
CREATE TABLE DrugInformation (
    drug_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    manufacturer_id INT NOT NULL,
    batch_number VARCHAR(100),
    expiry_date DATE,
    type_id INT NOT NULL,
    current_version INT DEFAULT 1,
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_update_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (manufacturer_id) REFERENCES ManufacturerInformation(manufacturer_id),
    FOREIGN KEY (type_id) REFERENCES DrugType(type_id)
);

-- 药品信息历史表
CREATE TABLE DrugInformationHistory (
    history_id INT AUTO_INCREMENT PRIMARY KEY,
    drug_id INT NOT NULL,
    version INT NOT NULL,
    name VARCHAR(255) NOT NULL,
    manufacturer_id INT NOT NULL,
    batch_number VARCHAR(100),
    expiry_date DATE,
    type_id INT NOT NULL,
    modified_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    modified_by INT,
    change_description TEXT,
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_update_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (drug_id) REFERENCES DrugInformation(drug_id),
    FOREIGN KEY (manufacturer_id) REFERENCES ManufacturerInformation(manufacturer_id),
    FOREIGN KEY (type_id) REFERENCES DrugType(type_id)
);

-- 药品类型表
CREATE TABLE DrugType (
    type_id INT AUTO_INCREMENT PRIMARY KEY,
    type_name VARCHAR(100) NOT NULL,
    description TEXT,
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_update_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 制造商信息表
CREATE TABLE ManufacturerInformation (
    manufacturer_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    address TEXT,
    contact_info VARCHAR(255),
    country VARCHAR(100),
    established_date DATE,
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_update_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 运输信息表
CREATE TABLE ShipmentInformation (
    shipment_id INT AUTO_INCREMENT PRIMARY KEY,
    drug_id INT NOT NULL,
    from_location VARCHAR(255),
    to_location VARCHAR(255),
    departure_date DATE,
    arrival_date DATE,
    carrier_id INT,
    status VARCHAR(50),
    tracking_number VARCHAR(100),
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_update_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (drug_id) REFERENCES DrugInformation(drug_id)
);

-- 存储信息表
CREATE TABLE StorageInformation (
    storage_id INT AUTO_INCREMENT PRIMARY KEY,
    drug_id INT NOT NULL,
    location_id INT,
    temperature DECIMAL(5, 2),
    humidity DECIMAL(5, 2),
    storage_date DATE,
    capacity INT,
    status VARCHAR(50),
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_update_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (drug_id) REFERENCES DrugInformation(drug_id)
);

-- 销售信息表
CREATE TABLE SalesInformation (
    sale_id INT AUTO_INCREMENT PRIMARY KEY,
    drug_id INT NOT NULL,
    retailer_id INT,
    sale_date DATE,
    quantity INT,
    price DECIMAL(10, 2),
    discount DECIMAL(5, 2),
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_update_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (drug_id) REFERENCES DrugInformation(drug_id)
);

-- 区块链交易信息表
CREATE TABLE BlockchainTransactionInformation (
    transaction_id INT AUTO_INCREMENT PRIMARY KEY,
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    previous_hash VARCHAR(255),
    current_hash VARCHAR(255),
    data TEXT,
    transaction_type VARCHAR(50),
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_update_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
